JavaScript-operators

Aerithmetic Operators (+, -, *, /, %, ++, --)

var a=10, b=10, ans=a+b; //ans=20 (+)
var a=10, b=20, ans=a-b; //ans=-10 (-)
var a=10, b=20, ans=a*b; //ans=200 (*)
var a=50, b=2, ans=a/b; //ans=25 (/)
var a=50, b=2, ans=a%b; //ans=0 (%)
var a=50, ans= ++a; //ans=51 (++)
var a=50, ans= --a; //ans=49 (--)

Assignment Operators (=, +=, -=, *=, /=, %= )

var a=30; // (=)
var a=10, b=30; // a += b //a = 40 (+=)
var a=10, b=30; // a -= b //a = -20 (-=)
var a=10, b=30; // a *= b //a = 300 (*=)
var a=30, b=10; // a /= b //a = 3 (/=)
var a=30, b=10; // a %= b //a = 0 (%=)

Comparison Operators (==, ===, <=, >=)

var a=10, b="10", ans=(a==b); //true (==)
var a=10, b="10", ans=(a===b); //false (===)
var a=10, b=20, ans=(a<=b); //true (<=)
var a=10, b=20, ans=(a>=b); //false (>=)

Logical Operators (&&, ||, ! )

var a=20, b=30, c=30; //a==b && b==c //false
var a=20, b=30, c=30; //a==b || b==c //true
var a=20, b=30; // !(a>b) //true
